// En AppointmentController.php

public function store(Request $request)
{
    $validator = Validator::make($request->all(), [
        'reservable_uuid' => 'required|uuid',
        'reservable_type' => 'required|string|in:App\\Models\\ConsultingRoom,App\\Models\\OperatingRoom,App\\Models\\MedicalTest',
        'appointmentDate' => 'required|date',
        'doctor_uuid' => 'required_if:reservable_type,App\\Models\\ConsultingRoom,App\\Models\\OperatingRoom',
        'names' => 'required|string',
        'first_last_name' => 'required|string',
        'second_last_name' => 'nullable|string',
        'phoneNumber' => 'required|string',
    ]);

    if ($validator->fails()) {
        return response()->json(['errors' => $validator->errors()], 422);
    }

    try {
        DB::transaction(function () use ($request) {
            $person = Person::where('user_uuid', Auth::user()->uuid)->first();
            if (!$person) {
                throw new \Exception('El usuario no existe.');
            }

            // Crear la reserva
            $reservationData = [
                'reservable_uuid' => $request->reservable_uuid,
                'reservable_type' => $request->reservable_type,
                'date' => $request->appointmentDate,
                'registered_by_uuid' => $person->uuid,
            ];

            // Solo agregar doctor_uuid si es necesario
            if ($request->doctor_uuid) {
                $reservationData['doctor_uuid'] = $request->doctor_uuid;
            }

            $reservation = Reservation::create($reservationData);

            // Crear la cita
            $appointment = new Appointment();
            $appointment->names = $request->names;
            $appointment->first_last_name = $request->first_last_name;
            $appointment->second_last_name = $request->second_last_name;
            $appointment->phone_number = $request->phoneNumber;
            $appointment->appointment_date = $request->appointmentDate;
            $appointment->registered_by_uuid = Auth::user()->uuid;
            $appointment->reservation_uuid = $reservation->uuid;
            $appointment->save();
        });

        return response()->json(['message' => 'Reserva creada con éxito.'], 201);
    } catch (\Exception $e) {
        return response()->json(['error' => $e->getMessage()], 400);
    }
}
